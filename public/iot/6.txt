#include <DHT.h>
#include <DHT_U.h>

#define DHTPIN 2     
#define DHTTYPE DHT11   // DHT 11
#define LIGHTPIN 13      // Pin to control the light

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  Serial.println(F("DHTxx test!"));

  dht.begin();
  pinMode(LIGHTPIN, OUTPUT);  // Set the light pin as an output
  digitalWrite(LIGHTPIN, LOW); // Ensure the light is off initially
}

void loop() {
  delay(2000);

  float h = dht.readHumidity();
  float t = dht.readTemperature();
  float f = dht.readTemperature(true);

  if (isnan(h) || isnan(t) || isnan(f)) {
    Serial.println(F("Failed to read from DHT sensor!"));
    return;
  }

  float hif = dht.computeHeatIndex(f, h);
  float hic = dht.computeHeatIndex(t, h, false);

  Serial.print(F("Humidity: "));
  Serial.print(h);
  Serial.print(F("%  Temperature: "));
  Serial.print(t);
  Serial.print(F("°C "));
  Serial.print(f);
  Serial.print(F("°F  Heat index: "));
  Serial.print(hic);
  Serial.print(F("°C "));
  Serial.print(hif);
  Serial.println(F("°F"));

  // Check conditions for turning on the light
  if (h > 50 || t > 30) {
    digitalWrite(LIGHTPIN, HIGH);

    // Turn the light on
  } else {
    digitalWrite(LIGHTPIN, LOW);  // Turn the light off
  }
}

#include <SimpleDHT.h>

// Pin configurations
int pinDHT11 = 8; // DHT11 data pin
int ledPin = 13;  // LED pin

SimpleDHT11 dht11(pinDHT11);

void setup() {
  pinMode(ledPin, OUTPUT); // Set the LED pin as output
  Serial.begin(9600);
  Serial.println("DHT11 Sensor with LED Control");
}

void loop() {
  byte temperature = 0;
  byte humidity = 0;

  // Read data from the DHT11 sensor
  int err = dht11.read(&temperature, &humidity, nullptr);
  if (err != SimpleDHTErrSuccess) {
    Serial.print("Read error: ");
    Serial.println(err);
    delay(2000); // Delay before retrying
    return;
  }

  // Print the temperature and humidity
  Serial.print("Temperature: ");
  Serial.print((int)temperature);
  Serial.println(" °C");

  Serial.print("Humidity: ");
  Serial.print((int)humidity);
  Serial.println(" %");

  // Check if temperature exceeds 30°C
  if (temperature > 30) {
    digitalWrite(ledPin, HIGH); // Turn on the LED
    Serial.println("Temperature is high! LED is ON.");
  } else {
    digitalWrite(ledPin, LOW); // Turn off the LED
    Serial.println("Temperature is normal. LED is OFF.");
  }

  // Wait before the next reading
  delay(3000);
}
